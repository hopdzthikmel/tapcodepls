-- Script để tự động quay unit trong Anime Vanguards

-- Tham chiếu đến các đối tượng cần thiết
local player = game.Players.LocalPlayer
local gemCost = 100 -- Số gems tiêu hao cho mỗi lượt quay
local rollInterval = 5 -- Khoảng thời gian giữa các lần quay (tính bằng giây)
local banner = game.Workspace:WaitForChild("Banner") -- Thay đổi đường dẫn theo cách trò chơi của bạn
local gems = player:WaitForChild("Gems") -- Thay đổi nếu gems được lưu ở vị trí khác

-- Hàm để thực hiện quay
local function rollForUnit()
    if gems.Value >= gemCost then
        -- Tiêu hao gems
        gems.Value = gems.Value - gemCost

        -- Thực hiện quay
        local rollEvent = banner:FindFirstChild("RollEvent") -- Thay đổi tên và phương thức gọi quay nếu khác
        if rollEvent then
            rollEvent:FireServer() -- Gọi sự kiện để thực hiện quay
        else
            warn("RollEvent not found in Banner")
            return
        end

        -- Chờ kết quả quay
        wait(1) -- Thay đổi thời gian chờ nếu cần

        -- Kiểm tra kết quả quay
        local result = player:FindFirstChild("RollResult") -- Thay đổi theo cách trò chơi cung cấp kết quả quay
        if result and result.Value == "Sung Jinwoo" then
            print("Received unit: Sung Jinwoo")
            -- Thực hiện hành động khi nhận được unit
        else
            print("Did not receive unit. Rolling again...")
        end
    else
        print("Not enough gems.")
    end
end

-- Vòng lặp chính để tự động quay
while true do
    rollForUnit()
    wait(rollInterval)
end
